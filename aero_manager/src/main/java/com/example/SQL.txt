CREATE TABLE Aviao (
    id INT PRIMARY KEY AUTO_INCREMENT,
    modelo VARCHAR(50) NOT NULL,
    capacidade INT NOT NULL
);

CREATE TABLE Aeroporto (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    codigo_iata VARCHAR(3) UNIQUE NOT NULL -- Ex: GRU, JFK
);

CREATE TABLE Voo (
    id INT PRIMARY KEY AUTO_INCREMENT,
    aviao_id INT NOT NULL,
    origem_id INT NOT NULL,
    destino_id INT NOT NULL,
    horario DATETIME NOT NULL,
    FOREIGN KEY (aviao_id) REFERENCES Aviao(id),
    FOREIGN KEY (origem_id) REFERENCES Aeroporto(id),
    FOREIGN KEY (destino_id) REFERENCES Aeroporto(id)
);

CREATE TABLE Passageiro (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    cpf VARCHAR(11) UNIQUE NOT NULL
);

CREATE TABLE Reserva (
    id INT PRIMARY KEY AUTO_INCREMENT,
    voo_id INT NOT NULL,
    passageiro_id INT NOT NULL,
    assento VARCHAR(10) NOT NULL,
    FOREIGN KEY (voo_id) REFERENCES Voo(id),
    FOREIGN KEY (passageiro_id) REFERENCES Passageiro(id)
);
